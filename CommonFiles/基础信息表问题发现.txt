--最新的装机省市提取方法
 SELECT DISTINCT A.AREA_NM 省,  A1.AREA_NM 市
   FROM BAP.T_BAP_MEC_IF B
   LEFT JOIN BAP.T_BAP_CDE_AREA A
     ON B.CUP_CODE = A.CUP_CODE
    AND A.AREA_TYP = '2'
   LEFT JOIN BAP.T_BAP_CDE_AREA A1
     ON A.AREA_ID = A1.PARENT_ID
    AND A1.AREA_TYP = '3'
  WHERE A.AREA_NM = '四川省';

三、结算基础表
标准秒到，打包秒到，即日付，时时付的结算信息无法提取  
原因：例如标准秒到需要根据结算类型和结算金额判断是否为标准秒到，是对应单笔的，而基础表中是汇总的一天的，无法判断单笔交易金额
                          Sum(Case
                                 When PAY_TYP = '12' And SET_FEE_AMT > 0 Then
                                  set_amt
                                 Else
                                  0
                               End) As JS13, --标准秒到
                           Sum(Case
                                 When PAY_TYP = '12' And SET_FEE_AMT = 0 Then
                                  set_amt
                                 Else
                                  0
                               End) As JS14, --打包秒到
                           Sum(Case
                                 When t0_typ = '02' And SET_FEE_AMT > 0 Then
                                  set_amt
                                 Else
                                  0
                               End) As JS15, --即日付
                           Sum(Case
                                 When t0_typ = '02' And SET_FEE_AMT = 0 Then
                                  set_amt
                                 Else
                                  0
                               End) As JS16, --时时付
