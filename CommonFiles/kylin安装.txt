1.环境变量配置
2.将hive-site.xml拷贝到$KYLIN_HOME/conf/下
3.将$HIVE_HOME/lib/下hive-jdbc-1.1.0-cdh5.11.0-standalone.jar，hive-jdbc-1.1.0-cdh5.11.0-standalone.jar
  包拷贝到$HADOOP_HOME/lib/下
4.修改kylin.properties
	### STORAGE ###
# The storage for cube is hbase
kylin.storage.url=hbase
# Compression codec for htable, valid value [none, snappy, lzo, gzip, lz4]
kylin.storage.hbase.compression-codec=snappy --修改压缩算法，数据量缩减一半
5.邮箱配置
	# If true, will send email notification on job complete
		kylin.job.notification-enabled=true
		kylin.job.notification-mail-enable-starttls=true
		kylin.job.notification-mail-host=mail.suixingpay.com
		kylin.job.notification-mail-port=25
		kylin.job.notification-mail-username=dc_mail@suixingpay.com
		kylin.job.notification-mail-password=datamail123
		kylin.job.notification-mail-sender=dc_mail@suixingpay.com
6.cube参数
	### CUBE | DICTIONARY ###
	# 'auto', 'inmem' or 'layer'
		kylin.cube.algorithm=layer
	# A smaller threshold prefers layer, a larger threshold prefers in-mem
		kylin.cube.algorithm.layer-or-inmem-threshold=7
		ylin.cube.aggrgroup.max-combination=40960000 --最大组合数
		kylin.snapshot.max-mb=3000 --快照大小
